class Solution {
    public boolean canJump(int[] nums) {
        if (nums.length == 1) {
            return true;
        } else if (nums[0] == 0) {
            return false;
        }

        boolean[] dp = new boolean[nums.length];
        dp[0] = true;
        dp[1] = true;
        for (int i = 0; i < nums.length - 1; i++) {
            if (!dp[i]) {
                continue;
            }
            for (int j = 0; j <= nums[i]; j++) {
                if (i + j < nums.length) {
                    dp[i + j] = true;
                }
            }
        }
        return dp[nums.length - 1];
    }
}

class Solution {
    public boolean canJump(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[n - 1] = 0;
        for (int i = n - 2; i >= 0; i--) {
            int j = nums[i];
            dp[i] = Math.max(getMax(dp, i + 1, i + j), i + j);
        }
        return dp[0] >= n - 1;
    }

    private int getMax(int[] dp, int start, int end) {
        int max = -1;
        for (int i = start; i <= end; i++) {
            if (i >= dp.length) {
                return max;
            }
            if (dp[i] > max) {
                max = dp[i];
            }
        }
        return max;
    }
}

class Solution {
    public boolean canJump(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[n - 1] = n - 1;
        for (int i = n - 2; i >= 0; i--) {
            int j = nums[i];
            if (j == 0) {
                dp[i] = i;
            } else {
                int k = nums[i + 1];
                if (i + j <= i + 1 + k) {
                    dp[i] = dp[i + 1];
                } else {
                    int tmp = getMax(dp, i + 1 + k + 1, i + j);
                    dp[i] = Math.max(Math.max(dp[i + 1], tmp), i + j);
                }
            }
        }
        return dp[0] >= n - 1;
    }

    private int getMax(int[] dp, int start, int end) {
        int max = -1;
        for (int i = start; i <= end; i++) {
            if (i >= dp.length) {
                return max;
            }
            if (dp[i] > max) {
                max = dp[i];
            }
        }
        return max;
    }
}

public boolean canJump(int[] nums) {
    int max = nums[0];
    int n = nums.length;
    for (int i = 1; i < n; i++) {
        if (i <= max) {
            max = Math.max(max, nums[i] + i);
            if (max >= n - 1) {
                return true;
            }
        }
    }
    return false;
}
