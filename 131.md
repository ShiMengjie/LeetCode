# [131. 分割回文串](https://leetcode-cn.com/problems/palindrome-partitioning/)







```java
public List<List<String>> partition(String s) {
    List<List<String>> res = new LinkedList<>();
    List<String> path = new LinkedList<>();
    backtrack(res, path, s, 0);
    return res;
}

private void backtrack(List<List<String>> res, List<String> path, String s, int idx) {
    if (idx >= s.length()) {
        res.add(new ArrayList<>(path));
        return;
    }
    int dif = s.length() - idx;
    for (int i = 1; i <= dif; i++) {
        if (isPalindrome(s, idx, idx + i - 1)) {
            path.add(s.substring(idx, idx + i));
            backtrack(res, path, s, idx + i);
            path.remove(path.size() - 1);
        }
    }
}

private boolean isPalindrome(String s, int start, int end) {
    while (start <= end) {
        if (s.charAt(start) != s.charAt(end)) {
            return false;
        }
        start++;
        end--;
    }
    return true;
}
```
