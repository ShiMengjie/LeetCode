# [188. 买卖股票的最佳时机 IV](https://leetcode-cn.com/problems/best-time-to-buy-and-sell-stock-iv/)





```java
class Solution {
    public int maxProfit(int k, int[] prices) {
        if(prices.length == 0) {
            return 0;
        }
        int n = prices.length;
        k = Math.min(k, n / 2);
        int[][] b = new int[n][k + 1], s = new int[n][k + 1];

        b[0][0] = -prices[0];
        s[0][0] = 0;
        int max = Math.max(b[0][0], s[0][0]);
        for (int i = 1; i <= k; i++) {
            b[0][i] = s[0][i] = Integer.MIN_VALUE / 2;
        }
        for (int i = 1; i < n; i++) {
            b[i][0] = Math.max(b[i - 1][0], s[i - 1][0] - prices[i]);
            for (int j = 1; j <= k; j++) {
                b[i][j] = Math.max(b[i - 1][j], s[i - 1][j] - prices[i]);
                s[i][j] = Math.max(s[i - 1][j], b[i - 1][j - 1] + prices[i]);
                int tmp = Math.max(b[i][j], s[i][j]);
                max = Math.max(max, tmp);
            }
        }
        return max;
    }
}
```
